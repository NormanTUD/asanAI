Definition OpenSet := { Selbst, Andere, Gemeinschaft }.

Definition Overlap (U V : OpenSet) : OpenSet :=
  match U, V with
  | Selbst, Andere => SelbstAndere    (* Empathie-Raum *)
  | Andere, Gemeinschaft => AndereGem  (* Sozial-Raum *)
  | Selbst, Gemeinschaft => SelbstGem  (* Verantwortungs-Raum *)
  | _, _ => Empty
  end.

Record EthicalSection := {
  achtsamkeit   : Prop;
  empathie      : Prop;
  wahrhaftigkeit: Prop;
  nichtschaden  : Prop;
  verantwortung : Prop;
  fairness      : Prop;
  fuersorge     : Prop;
  maessigung    : Prop;
  wachstum      : Prop;
}.

Definition Restriction (s : EthicalSection) (U : OpenSet) : EthicalSection :=
  match U with
  | SelbstAndere => {|
      achtsamkeit := s.(achtsamkeit);
      empathie := s.(empathie);
      wahrhaftigkeit := s.(wahrhaftigkeit);
      nichtschaden := True;
      verantwortung := True;
      fairness := True;
      fuersorge := True;
      maessigung := True;
      wachstum := True;
    |}
  | _ => s
  end.

Axiom Achtsamkeit_universal :
  forall (U : OpenSet), (Restriction sectionSelbst U).(achtsamkeit) =
                         (Restriction sectionAndere U).(achtsamkeit).

Axiom Empathie_universal :
  forall (U : OpenSet), (Restriction sectionSelbst U).(empathie) =
                         (Restriction sectionAndere U).(empathie).

Axiom Wahrhaftigkeit_universal :
  forall (U : OpenSet), (Restriction sectionSelbst U).(wahrhaftigkeit) =
                         (Restriction sectionAndere U).(wahrhaftigkeit).

(* Kategorientheoretische Struktur *)
Record Category := {
  Obj : Type;
  Hom : Obj -> Obj -> Type;
  id  : forall A, Hom A A;
  comp: forall A B C, Hom A B -> Hom B C -> Hom A C;
}.

Record Functor (C D : Category) := {
  fobj : C.(Obj) -> D.(Obj);
  fmap : forall A B, C.(Hom) A B -> D.(Hom) (fobj A) (fobj B);
  fmap_id : forall A, fmap A A (C.(id) A) = D.(id) (fobj A);
  fmap_comp : forall A B C' (f: C.(Hom) A B) (g: C.(Hom) B C'),
    fmap A C' (C.(comp) A B C' f g) = D.(comp) (fobj A) (fobj B) (fobj C') (fmap A B f) (fmap B C' g);
}.

Record NatTrans (C D : Category) (F G : Functor C D) := {
  component : forall A : C.(Obj), D.(Hom) (F.(fobj) A) (G.(fobj) A);
  naturality : forall (A B : C.(Obj)) (f : C.(Hom) A B),
    D.(comp) _ _ _ (F.(fmap) A B f) (component B) =
    D.(comp) _ _ _ (component A) (G.(fmap) A B f);
}.

(* Homotopietyptheorie: Pfade als Wachstum *)
Definition Path (A : Type) (x y : A) := x = y.
Definition HigherPath (A : Type) (x y : A) (p q : Path A x y) := p = q.

(* Garbenstruktur *)
Record Sheaf := {
  sections : OpenSet -> EthicalSection;
  gluing : forall (U V : OpenSet),
    Restriction (sections U) (Overlap U V) =
    Restriction (sections V) (Overlap U V);
  locality : forall (U : OpenSet) (s t : EthicalSection),
    (forall V, Restriction s (Overlap U V) = Restriction t (Overlap U V)) -> s = t;

Ohne Bewusstsein für den gegenwärtigen Moment ist keine ethische Entscheidung möglich. Dies ist die Nullbedingung: Bevor wir gut handeln können, müssen wir überhaupt handeln — und dafür müssen wir wach sein.

Empathie als Perspektivwechsel

Empathie ist nicht Mitleid — es ist die kognitive Fähigkeit, die Welt aus der Perspektive eines anderen zu sehen. In der Kategorientheorie: ein Funktor, der Objekte und Morphismen der eigenen Kategorie (Selbst) in die Kategorie des Anderen abbildet, strukturerhaltend.

Konvergenz: Konfuzianisches Ren (仁), Ubuntu ("Ich bin, weil wir sind"), Bubers "Ich und Du", Adam Smiths "Theory of Moral Sentiments", Levinas' Antlitz des Anderen.

Quellen: Konfuzius, Analekten 12.2; Martin Buber, "Ich und Du" (1923); Emmanuel Levinas, "Totalité et Infini" (1961); Adam Smith, "Theory of Moral Sentiments" (1759)
Kategorientheoretisch: Empathie als Funktor

Sei Selbst die Kategorie der eigenen Erfahrungen und Andere die Kategorie der Erfahrungen des Anderen.

Definition: Empathie E : Selbst → Andere ist ein Funktor mit:
E(obj) : Erfahrung_Selbst → Erfahrung_Andere (Objektabbildung)
E(mor) : Beziehung_Selbst → Beziehung_Andere (Morphismenabbildung)

Funktorbedingungen:
1. E(id_A) = id_{E(A)} — Identität bleibt erhalten
2. E(g ∘ f) = E(g) ∘ E(f) — Komposition bleibt erhalten

Garbentheoretisch: Empathie ist eine Sektion auf Overlap(Selbst, Andere)
s_empathie ∈ F(Selbst ∩ Andere)

✓ Sub-H2.1: Funktor E erhält Identitäten (Selbstrespekt bleibt)
✓ Sub-H2.2: Funktor E erhält Kompositionen (Beziehungsketten übertragbar)
✓ Sub-H2.3: Empathie ist Brücken-Sektion auf Selbst∩Andere
✓ Sub-H2.4: Restriktion von Selbst auf Überlappung = Restriktion von Andere auf Überlappung

Nicht-Schaden als ethische Nullsektion

Primum non nocere — zuerst nicht schaden. Dies ist die minimale ethische Anforderung, die untere Schranke. In jeder ethischen Tradition existiert diese Grenze: Das Verbot zu schaden kommt vor dem Gebot zu helfen.

Konvergenz: Hippokratischer Eid, Buddhistische Ahimsa, Jainistisches Nicht-Verletzen, Kantisches "Handle so, dass...", Mills Harm Principle, Rawls' Grundsatz.

Quellen: Hippokrates; Patanjalis Yoga Sutras 2.35; Kant, "Grundlegung zur Metaphysik der Sitten" (1785); J.S. Mill, "On Liberty" (1859)
Garbentheoretisch: Nicht-Schaden als Nullsektion (untere Schranke)

Definition: Die Nullsektion 0 ∈ F(U) ist definiert als:
0(x) = "keine negative Wirkung auf x" für alle x ∈ U

Ordnungsstruktur: Für alle Sektionen s ∈ F(U) gilt:
0 ≤ s (Nicht-Schaden ist die untere Schranke)

Formal: Die ethische Garbe F hat eine partielle Ordnung ≤ auf ihren Sektionen, und die Nullsektion ist das Infimum:
∀ U offen, ∀ s ∈ F(U): 0_U ≤ s

Kategorientheoretisch: Nicht-Schaden ist der terminale Morphismus in die Null-Kategorie:
∀ Handlung H: schaden(H) = 0 ist notwendige Bedingung für ethisch(H) = true

✓ Sub-H3.1: Nullsektion existiert auf jeder offenen Menge
✓ Sub-H3.2: Nullsektion ist untere Schranke (∀s: 0 ≤ s)
✓ Sub-H3.3: Nullsektion ist kompatibel mit Restriktionen
✓ Sub-H3.4: Nicht-Schaden ist notwendige Vorbedingung für Hilfe (Schritt 7)

Wahrhaftigkeit als Lokalitätsbedingung

Wahrhaftigkeit ist die Bedingung, dass lokale Information nicht verfälscht wird. In der Garbentheorie: Eine Sektion, die lokal korrekt ist, muss auch global konsistent sein. Lügen zerstören die Verklebungsbedingung — sie machen lokale Sektionen inkompatibel.

Konvergenz: Kantisches Lügenverbot, Buddhistisches Rechtes Reden, Konfuzianisches Zhengming (正名 — Richtigstellung der Namen), Habermas' kommunikative Rationalität, Petersons "Tell the truth, or at least don't lie."

Quellen: Kant, "Über ein vermeintes Recht aus Menschenliebe zu lügen" (1797); Habermas, "Theorie des kommunikativen Handelns" (1981); Bok, "Lying" (1978)
Garbentheoretisch: Wahrhaftigkeit als Lokalitätsbedingung

Erinnerung: Die Lokalitätsbedingung einer Garbe F besagt:
Wenn zwei Sektionen s, t ∈ F(U) auf allen offenen Teilmengen übereinstimmen, dann sind sie gleich:
(∀ V ⊆ U offen: s|_V = t|_V) → s = t

Interpretation: Wahrhaftigkeit IST die Lokalitätsbedingung. Wenn du lokal (in jeder Situation) die Wahrheit sagst, dann ist dein globales Bild konsistent. Lügen erzeugen Sektionen, die lokal inkonsistent sind — die Garbe "verklebt nicht."

Formal:
Wahrhaftigkeit(s) := ∀ U ⊆ X offen: s|_U spiegelt die tatsächliche lokale Struktur wider
¬Wahrhaftigkeit(s) → ∃ U, V: s|_{U∩V} von U ≠ s|_{U∩V} von V (Verklebung bricht)

✓ Sub-H4.1: Wahrhaftigkeit impliziert Lokalität der Garbe
✓ Sub-H4.2: Lüge impliziert Verletzung der Verklebungsbedingung
✓ Sub-H4.3: Wahrhaftigkeit ist Brückenkonzept (auf Selbst∩Andere definiert)
✓ Sub-H4.4: Wahrhaftigkeit ist kompatibel mit Nicht-Schaden (keine destruktive Wahrheit nötig)

Verantwortung als Kausal-Funktor

Verantwortung ist die Zuordnung von Wirkungen zu ihren Ursachen — und die Bereitschaft, diese Zuordnung anzuerkennen. Kategorientheoretisch: ein kovarianter Funktor von der Kategorie der Handlungen in die Kategorie der Konsequenzen, der die kausale Struktur bewahrt.

Konvergenz: Existentialistische Verantwortung (Sartre: "Der Mensch ist zur Freiheit verurteilt"), Karma-Konzept, Stoische Eigenverantwortung, Hans Jonas' "Prinzip Verantwortung".

Quellen: Sartre, "L'existentialisme est un humanisme" (1946); Jonas, "Das Prinzip Verantwortung" (1979); Aristoteles, Nikomachische Ethik III
Kategorientheoretisch: Verantwortung als kovarianter Funktor

Definition: Sei Hand die Kategorie der Handlungen und Kons die Kategorie der Konsequenzen.
Verantwortung V : Hand → Kons ist ein kovarianter Funktor:

V(handlung) = konsequenz (Objektabbildung)
V(h₁ → h₂) = V(h₁) → V(h₂) (Morphismenabbildung, kovariant)

Kovariant bedeutet: Die Richtung bleibt erhalten. Wer handelt, trägt die Konsequenz in derselben Richtung.

Garbentheoretisch: Verantwortung ist die Sektion, die auf Selbst∩Gemeinschaft definiert ist — sie verbindet individuelle Handlung mit kollektiver Wirkung.

✓ Sub-H5.1: Funktor V ist kovariant (Richtungserhaltung)
✓ Sub-H5.2: V erhält Identitäten (Nichtstun → keine Konsequenz)
✓ Sub-H5.3: V erhält Kompositionen (Handlungsketten → Konsequenzketten)
✓ Sub-H5.4: Sektion auf Selbst∩Gemeinschaft ist wohldefiniert

Gerechtigkeit als Symmetrie / Natürliche Transformation

Gerechtigkeit ist die Forderung nach Symmetrie: Was für mich gilt, muss auch für dich gelten (unter gleichen Bedingungen). In der Kategorientheorie ist dies eine natürliche Transformation — eine strukturerhaltende Abbildung zwischen zwei Funktoren, die "natürlich" (d.h. unabhängig von der Wahl des Objekts) ist.

Konvergenz: Goldene Regel (in allen Kulturen), Rawls' Schleier des Nichtwissens, Kantischer Kategorischer Imperativ, Konfuzianisches Shu (恕 — Gegenseitigkeit).

Quellen: Rawls, "A Theory of Justice" (1971); Kant, KpV; Konfuzius, Analekten 15.24; Levitikus 19:18
Kategorientheoretisch: Gerechtigkeit als natürliche Transformation

Seien F, G : Personen → EthischeAnsprüche zwei Funktoren.
F = "was ich für mich beanspruche"
G = "was ich dem anderen zugestehe"

Definition: Gerechtigkeit η : F ⇒ G ist eine natürliche Transformation:
∀ Person P: η_P : F(P) → G(P)

Natürlichkeitsbedingung:
∀ f : P → Q (Beziehung zwischen Personen):
G(f) ∘ η_P = η_Q ∘ F(f)

Bedeutung: Egal welchen "Weg" man nimmt (erst transformieren dann abbilden, oder erst abbilden dann transformieren) — das Ergebnis ist dasselbe. Das IST Fairness: Die Reihenfolge spielt keine Rolle, die Struktur ist symmetrisch.

Garbentheoretisch: Gerechtigkeit ist die Symmetriebedingung:
∀ U offen: s|_U(Person A) = s|_U(Person B) wenn Kontext(A) = Kontext(B)

✓ Sub-H6.1: Natürliche Transformation η existiert
✓ Sub-H6.2: Natürlichkeitsbedingung (Kommutativität) ist erfüllt
✓ Sub-H6.3: Symmetrie impliziert Goldene Regel
✓ Sub-H6.4: Kompatibel mit Empathie-Funktor (Schritt 2)

Wachstum ist kein Sprung — es ist ein kontinuierlicher Pfad von einem Zustand zu einem besseren. In der Homotopietyptheorie: ein Pfad zwischen zwei Punkten im Raum der ethischen Zustände. Und Meta-Lernen (lernen zu lernen) ist ein höherer Pfad — ein Pfad zwischen Pfaden.

Homotopietyptheorie: Wachstum als Pfad

Definition: Sei EthState der Typ der ethischen Zustände.

Höhere Pfade (Meta-Lernen):
Sei p, q : Path EthState s t zwei verschiedene Wachstumswege.
Ein höherer Pfad h : Path (Path EthState s t) p q ist ein Beweis, dass man zwischen Wachstumswegen wechseln kann — Meta-Lernen.
