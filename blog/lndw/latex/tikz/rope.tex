% ── Farben ──
\definecolor{dim1}{HTML}{E8590C}
\definecolor{dim2}{HTML}{0F3460}
\definecolor{dim3}{HTML}{E94560}
\definecolor{dim4}{HTML}{1A8F3C}
\definecolor{dim5}{HTML}{7B2D8B}
\definecolor{dim6}{HTML}{D4A017}

% ── Frequenzen: θ_i = 1/10000^(2i/d) ──
% Für Visualisierung skaliert (sonst zu klein sichtbar)
% Innerer Kreis = schnellste Rotation, äußerer = langsamste
\pgfmathsetmacro{\freqA}{36}    % θ_0 · Skalierung → 36° pro Position
\pgfmathsetmacro{\freqB}{24}    % θ_1
\pgfmathsetmacro{\freqC}{16}    % θ_2
\pgfmathsetmacro{\freqD}{10}    % θ_3
\pgfmathsetmacro{\freqE}{6}     % θ_4
\pgfmathsetmacro{\freqF}{3}     % θ_5 → langsamste

% ── Winkel berechnen: angle = freq * pos ──
\pgfmathsetmacro{\angleA}{\freqA * \pos}
\pgfmathsetmacro{\angleB}{\freqB * \pos}
\pgfmathsetmacro{\angleC}{\freqC * \pos}
\pgfmathsetmacro{\angleD}{\freqD * \pos}
\pgfmathsetmacro{\angleE}{\freqE * \pos}
\pgfmathsetmacro{\angleF}{\freqF * \pos}

% ── Radien ──
\def\rA{1.3}
\def\rB{2.1}
\def\rC{2.9}
\def\rD{3.7}
\def\rE{4.5}
\def\rF{5.3}

% ══════════════════════════════════════════════════════════════
%  HINTERGRUND: Leichter radialer Gradient-Effekt
% ══════════════════════════════════════════════════════════════
\fill[white] (-7.5,-7.5) rectangle (7.5,7.5);

% Feine Referenzlinien (0°-Achse = Position 0)
\draw[gray!20, thin] (0,0) -- (0:\rF+0.5);
\draw[gray!20, thin] (0,0) -- (90:\rF+0.5);
\draw[gray!20, thin] (0,0) -- (180:\rF+0.5);
\draw[gray!20, thin] (0,0) -- (270:\rF+0.5);

% ══════════════════════════════════════════════════════════════
%  KREISE (Dimensionspaare) – mit Glow-Effekt
% ══════════════════════════════════════════════════════════════

% Kreis 6 (äußerster, langsamster)
\draw[dim6, thick, opacity=0.3] (0,0) circle (\rF);
\draw[dim6, line width=3pt, opacity=0.08] (0,0) circle (\rF);
\node[font=\tiny, dim6, opacity=0.7] at (170:\rF+0.4) {$(d_{10}, d_{11})$};

% Kreis 5
\draw[dim5, thick, opacity=0.35] (0,0) circle (\rE);
\draw[dim5, line width=3pt, opacity=0.08] (0,0) circle (\rE);
\node[font=\tiny, dim5, opacity=0.7] at (170:\rE+0.4) {$(d_8, d_9)$};

\draw[dim4, line width=3pt, opacity=0.08] (0,0) circle (\rD);

% Kreis 2

\draw[dim2, line width=3pt, opacity=0.1] (0,0) circle (\rB);
\node[font=\tiny, dim2, opacity=0.9] at (170:\rB+0.4) {$(d_2, d_3)$};

% Kreis 1 (innerster, schnellster)
\draw[dim1, thick, opacity=0.7] (0,0) circle (\rA);
\draw[dim1, line width=3pt, opacity=0.12] (0,0) circle (\rA);
\node[font=\tiny, dim1] at (170:\rA+0.4) {$(d_0, d_1)$};

% ══════════════════════════════════════════════════════════════
%  ROTIERENDE PUNKTE – die eigentliche Position-Kodierung
% ══════════════════════════════════════════════════════════════

% Spur (Trail) – zeigt woher der Punkt kam
\ifnum\pos>0
  \pgfmathsetmacro{\prevAngleA}{\freqA * (\pos - 1)}
  \draw[dim1, thick, opacity=0.3, -{Stealth[length=2mm]}] 
    (\prevAngleA:\rA) arc (\prevAngleA:\angleA:\rA);
  \pgfmathsetmacro{\prevAngleB}{\freqB * (\pos - 1)}
  \draw[dim2, thick, opacity=0.3, -{Stealth[length=2mm]}] 
    (\prevAngleB:\rB) arc (\prevAngleB:\angleB:\rB);
  \pgfmathsetmacro{\prevAngleC}{\freqC * (\pos - 1)}
  \draw[dim3, thick, opacity=0.3, -{Stealth[length=2mm]}] 
    (\prevAngleC:\rC) arc (\prevAngleC:\angleC:\rC);
  \pgfmathsetmacro{\prevAngleD}{\freqD * (\pos - 1)}
  \draw[dim4, thick, opacity=0.3, -{Stealth[length=2mm]}] 
    (\prevAngleD:\rD) arc (\prevAngleD:\angleD:\rD);
  \pgfmathsetmacro{\prevAngleE}{\freqE * (\pos - 1)}
  \draw[dim5, thick, opacity=0.3, -{Stealth[length=2mm]}] 
    (\prevAngleE:\rE) arc (\prevAngleE:\angleE:\rE);
  \pgfmathsetmacro{\prevAngleF}{\freqF * (\pos - 1)}
  \draw[dim6, thick, opacity=0.3, -{Stealth[length=2mm]}] 
    (\prevAngleF:\rF) arc (\prevAngleF:\angleF:\rF);
\fi
