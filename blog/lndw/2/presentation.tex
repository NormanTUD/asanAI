  \begin{center}
    \small Jede Zahl $t$ wird auf einen \textbf{Winkel} $\omega_k \cdot t$ auf dem Einheitskreis abgebildet.\\
    Addition $\to$ Winkel addieren $\to$ Rotation!
  \end{center}
\end{frame}

\begin{frame}{Die Formel -- \"Uberblick}
  \formelbox{
    \text{Logit}(c \mid a, b) = \sum_{k \in \mathcal{K}} \alpha_k \cdot \cos\!\left(\frac{2\pi k\,(a + b - c)}{113}\right)
  }

  \bigskip
  \begin{itemize}
    \stich{$\mathcal{K}$ = wenige ``Schl\"usselfrequenzen'' (z.\,B.\ 5 St\"uck)}
    \stich{$\alpha_k$ = gelernte Amplitude}
    \stich{Wenn $c = (a+b) \bmod 113$: alle Cosinus $= 1$ $\Rightarrow$ \textbf{Maximum}}
    \stich{Wenn $c \neq (a+b) \bmod 113$: verschiedene Richtungen $\Rightarrow$ \textbf{Ausl\"oschung}}
  \end{itemize}

  \bigskip
  \begin{center}
    \colorbox{highlight!10}{%
      \parbox{0.6\textwidth}{\centering\small
        \textbf{Konstruktive Interferenz} am richtigen Ergebnis,\\
        \textbf{destruktive Interferenz} \"uberall sonst.
      }%
    }
  \end{center}
\end{frame}

\begin{frame}{Schritt 1: Embedding -- Zahlen werden zu Kreispunkten}
  \begin{columns}[T]
    \column{0.55\textwidth}
    \begin{itemize}
      \stich{Jede Zahl $t \in \{0, \ldots, 112\}$ wird eingebettet}
      \stich{Die Embedding-Matrix $W_E$ ist \textbf{sparse im Fourier-Raum}}
      \stich{Effektiv kodiert sie:\\[3pt]
             $\cos(\omega_k \cdot t)$ und $\sin(\omega_k \cdot t)$\\[3pt]
             f\"ur wenige Frequenzen $k$}
      \stich{$\omega_k = \frac{2\pi k}{113}$}
    \end{itemize}

    \column{0.4\textwidth}

\begin{frame}{Schritt 2: Attention -- Information zusammenf\"uhren}
  \begin{center}

  \bigskip
  \begin{itemize}
    \stich{Attention bewegt Fourier-Komponenten von $a$ und $b$ zur Position ``$=$''}
    \stich{Softmax \"uber 2 Positionen $\approx$ Sigmoid $\approx$ \textbf{linear}}
    \stich{Erzeugt \textbf{Produkte}: $\cos(\omega_k a) \cdot \cos(\omega_k b)$, etc.}
    \stich{Das sind genau die Terme, die man f\"ur die Additionsformel braucht!}
  \end{itemize}
\end{frame}

\begin{frame}{Schritt 3: MLP -- die trigonometrische Identit\"at}
  \formelbox{
    \cos(\omega_k(a+b)) = \cos(\omega_k a)\cos(\omega_k b) - \sin(\omega_k a)\sin(\omega_k b)
  }

  \bigskip
  \begin{itemize}
    \stich{Die 512 ReLU-Neuronen berechnen \textbf{kollektiv} diese Identit\"at}
    \stich{ReLU = st\"uckweise linear $\Rightarrow$ kann Produkte approximieren}
    \stich{Ergebnis: $\cos(\omega_k(a+b))$ und $\sin(\omega_k(a+b))$}
    \stich{\"Uber 90\% der Varianz erkl\"art!}
  \end{itemize}

  \bigskip
  \begin{center}
    \colorbox{akzent!10}{%
      \parbox{0.7\textwidth}{\centering\small
        \textbf{Aha!} Das Netz hat die Additionsformel aus der Schule\\
        \emph{selbst\"andig entdeckt} -- ohne sie je gesehen zu haben.
      }%
    }
  \end{center}
\end{frame}

\begin{frame}{Schritt 4: Unembedding -- Ablesen des Ergebnisses}
  \begin{itemize}
    \stich{Die Neuron-Logit-Matrix $W_L = W_U W_{\text{out}}$ ist \textbf{approximativ Rang 10}}
    \stich{5 Frequenzen $\times$ 2 (cos + sin) = 10 Richtungen}
    \stich{F\"ur jedes m\"ogliche Ergebnis $c$ wird berechnet:}
  \end{itemize}

  \formelbox{
    \text{Logit}(c) \;=\; \sum_{k \in \mathcal{K}} \alpha_k \Big[\cos(\omega_k c) \cdot \cos(\omega_k(a{+}b)) + \sin(\omega_k c) \cdot \sin(\omega_k(a{+}b))\Big]
  }

  \bigskip
  \begin{center}
    $= \displaystyle\sum_{k \in \mathcal{K}} \alpha_k \cos\!\big(\omega_k(a+b-c)\big)$
    \qquad {\small (Cosinus-Subtraktionsidentit\"at)}
  \end{center}
\end{frame}

    \stich{5 Frequenzen reichen f\"ur $>99{,}99\%$ Genauigkeit}
  \end{itemize}
\end{frame}
