\draw[->,very thick,accent2] (gut) -- (nichtgut) node[midway,left,font=\tiny] {Was passiert};

% Explanation
\draw[<-,thin,gray] (nichtgut.south east) -- (2.5,0) node[right,font=\small,align=left] {Attention verschiebt ``gut''\\nur ein \textbf{bisschen} Richtung\\Negation -- nicht bis\\zum Gegenteil!};

\end{center}
\vspace{0.3cm}
\begin{alertblock}{Aha!}
``nicht'' verschiebt den Punkt von ``gut'' nur leicht -- er bleibt in der \emph{Nähe} von ``gut''! Deshalb verwechseln LLMs manchmal Negationen.
\end{alertblock}
\end{frame}

\begin{frame}{Warum ist das so?}
\begin{itemize}
\item \textbf{Attention} kann Punkte nur als \textbf{gewichtete Summe} anderer Punkte verschieben
\item ``nicht'' + ``gut'' = irgendwo \emph{zwischen} den beiden Punkten
\item Es gibt kein ``springe direkt zum Gegenteil''-Modul!
\item Das Netzwerk muss über \textbf{mehrere Layer} hinweg lernen, Negation zu verarbeiten
\end{itemize}
